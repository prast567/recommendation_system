def get_top_articles(n, df=df):
    """
    INPUT:
    n - (int) the number of top articles to return
    df - (pandas dataframe) df as defined at the top of the notebook 
    
    OUTPUT:
    top_articles - (list) A list of the top 'n' article titles 
    
    """
    top_articles = []
    # Group df by article id take count and sort in descending order
    group_by_article = df.groupby('article_id').count().sort_values('user_id', ascending=False).head(n)
    
    # Add article titles to the list
    for item in sorted(list(group_by_article.index), reverse=True):
        top_articles.append(df[df['article_id'] == item]['title'].iloc[0])
        
    return top_articles # Return the top article titles from df

def find_similar_users(user_id, user_item=user_item, include_similarity=False):
    """
    INPUT:
    user_id - (int) a user_id
    user_item - (pandas dataframe) matrix of users by articles: 
                1's when a user has interacted with an article, 0 otherwise
    include_similarity - (bool) whether to include the similarity in the output
    
    OUTPUT:
    similar_users - (list) an ordered list where the closest users (largest dot product users)
                    are listed first
    
    Description:
    Computes the similarity of every pair of users based on the dot product
    Returns an ordered list of user ids. If include_similarity is True, returns a list of lists
    where the first element is the user id and the second the similarity.
    
    """
    

    
    
    # compute similarity of each user to the provided user
    matching_multiplication = user_item*user_item.loc[user_id]
    similarity = matching_multiplication.sum(axis=1)

    # sort by similarity
    similarity_sorted = similarity.sort_values(ascending=False)

    # remove the own user's id
    similarity_sorted.drop(user_id, inplace = True)
    #print(similarity_sorted)
    # create list of just the ids
    most_similar_users = similarity_sorted.index.to_list()

    # create list of just the similarities
    if include_similarity:
        similarity_val_list = []
        for item in similarity_sorted:
            similarity_val_list.append(item)
        similar_user_and_similarity = list(map(lambda x,y: [x,y], most_similar_users,similarity_val_list))
        return similar_user_and_similarity # return a list of lists with [[similar_user_id, similarity], ...]
    
    return most_similar_users # return a list of the users in order from most to least similar

def get_top_sorted_users(user_id, user_item=user_item):
    """
    INPUT:
    user_id - (int)
    user_item - (pandas dataframe) matrix of users by articles: 
            1's when a user has interacted with an article, 0 otherwise
    
            
    OUTPUT:
    neighbors_df - (pandas dataframe) a dataframe with:
                    neighbor_id - is a neighbor user_id
                    similarity - measure of the similarity of each user to the provided user_id
                    num_interactions - the number of articles viewed by the user
                    
    Other Details - sort the neighbors_df by the similarity and then by number of interactions where 
                    highest of each is higher in the dataframe, i.e. Descending order
    """
    
    # Your code here
    # Hint: find similar users, but include their similarity, from there we can find the number of interactions for each neighbor
    # Find similar user_ids and similarity
    
    user_article_matrix = pd.get_dummies(df[['article_id', 'user_id']], columns=["article_id"]) 
    user_article_matrix = user_article_matrix.set_index('user_id')
    user_item_temp = user_article_matrix.groupby('user_id').apply(lambda x: x.sum())
    
    
    similar_user_and_similarity = find_similar_users(user_id, include_similarity= True)
    
    # Create a nested list of [[neighbor_id, similarity, num_interactions]]
    similaruser_similarity_interation = []
    for item in similar_user_and_similarity:
        similaruser_similarity_interation.append([item[0], item[1], user_item_temp.loc[item[0]].sum(axis=0)])
    
    # Convert nested list to the neighbors_df dataFrame
    neighbors_df = pd.DataFrame(similaruser_similarity_interation,columns=['neighbor_id', 'similarity', 'num_interactions'])
    # Sort based on similary and num_interactions
    neighbors_df = neighbors_df.sort_values(['similarity', 'num_interactions',], ascending=[False, False])
    neighbors_df.reset_index(inplace=True)

    return neighbors_df # Return the dataframe specified in the doc_string

num_latent_feats = np.arange(10, 700+10, 20)
metric_scores = []

for k in num_latent_feats:
    # restructure with k latent features
    u_new, vt_new = u[:, :k], v[:k, :]
    
    # take dot product
    user_item_est = abs(np.around(np.dot(u_new, vt_new))).astype(int)
    # make sure the values are between 0 and 1
    user_item_est = np.clip(user_item_est, 0, 1)

    acc = accuracy_score(user_item.values.flatten(), user_item_est.flatten())

    recall = recall_score(user_item.values.flatten(), user_item_est.flatten())

def get_svd_similar_article_ids(article_id, vt, user_item=user_item, include_similarity=False):
    """
    INPUT:
    article_id - (int) an article id
    vt - (numpy array) vt matrix from SVD
    user_item - (pandas dataframe) matrix of users by articles: 
                1's when a user has interacted with an article, 0 otherwise
    include_similarity - (bool) whether to include the similarity in the output
    
    OUTPUT:
    article_ids - (list) a list of article ids that are in the same title cluster
    
    Description:
    Returns a list of the article ids similar using SVD factorization
    """
    
    most_similar_items = []
    # Find the index of the article_id
    article_id_column_name = 'article_id_' + str(article_id)
    article_idx = user_item.columns.get_loc(article_id_column_name) 

    # Find the cosine similarity of all articles

    # Hint: vt should be transposed before passing to cosine_similarity to get a 714 x 714 matrix 
    cos_sim = cosine_similarity(vt.T)
    # Get similarities only for the cos_sim of the article_idx
    article_similarities = cos_sim[article_idx]


    # Sort and return the articles, don't include the own article
    article_similarities_indices = sorted(range(len(article_similarities)), key=article_similarities.__getitem__, reverse=True)

    for item in article_similarities_indices:
        if item != article_idx:
            most_similar_items.append((user_item.columns[item], article_similarities[item]))
    
    if include_similarity:
        return most_similar_items # return a list of lists with [[similar_article_id, similarity], ...]
    
    most_similar_items = [int(article_id.strip('article_id_')) for article_id, _ in most_similar_items]
    return most_similar_items
